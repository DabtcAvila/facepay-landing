üíª Implementation

Simple to implement with automatic initialization and customizable settings.

// HTML
<button class="btn btn-primary btn-magnetic" data-magnetic-indicator>
    Magnetic Button
</button>

// JavaScript - Auto-initialized or custom
const magneticButtons = initMagneticButtons({
    magneticRadius: 140,
    magneticStrength: 0.5,
    dampening: 0.8,
    stiffness: 0.12,
    rippleDuration: 800,
    touchFeedback: true,
    throttleFPS: 60
});

// Add magnetic behavior dynamically
magneticButtons.addMagneticButton('.my-custom-button');

// Performance stats
console.log(magneticButtons.getStats());
